Recipes and Recipe Books

Recipe Namespace

# ------
# Recipe
# ------

Namespace: www.kocinando.com/receta.stxt

	Recipe:
	    Recipe Name: (1)
	    Description: (?) TEXT
	    Ingredients: (1)
			Ingredient: (+)
	    		Quantity: (1) NUMBER
	    		Units: (1)
	    		Name: (1)
	    Steps: (1)
	    	Step: (+) NATURAL
	    	Description: (1)
		Difficulty: (?) ENUM
		   	: easy
		   	: medium
	    	: hard
	    Calories: (?) NUMBER
	    Servings: (?) NUMBER
	    Document: (*) (www.dokumentando.com/doc_simple.stxt)
